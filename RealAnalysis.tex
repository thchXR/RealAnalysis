\section{预备知识：数学分析}

\subsection{数列的上、下极限}

\subsubsection{上、下确界的定义}

粗略地讲，对于$A \subseteq \R$，$A$ 的上确界即为$A$的最小上界,记作
\[
\sup A
\]
$A$的下确界为$A$的最大下界，记作
\[
\inf A
\]
\smallskip

% 数学分析九大定理之一，作为定理使用
\subsubsection{单调有界序列必收敛}
设$\{a_n \}_{n\ge 1}$为单调数列，现在对其单调性进行分类讨论：
\ITEMS
	\item 当$\{a_n \}_{n\ge 1}$单调递增时：若数列有界，则数列极限为上确界，即$\lim\limits_{n\to\infty} a_n = \sup\{a_n\}$；若数列无界，数列极限不存在，即$\lim\limits_{n\to\infty} a_n = +\infty$。
	\item 当$\{a_n \}_{n\ge 1}$单调递减时：若数列有界，则数列极限为下确界，即$\lim\limits_{n\to\infty} a_n = \inf\{a_n\}$；若数列无界，数列极限不存在，即$\lim\limits_{n\to\infty} a_n = -\infty$。
\ITEME

\smallskip

\subsubsection{上、下极限的定义}

\no {\bf 定义1}

设$\{a_n \}_{n\ge 1}$为一数列，其所有极限点（收敛子列的极限） 的上确界定义为上极限，记作
\[
	\varlimsup _{n\rightarrow \infty }a_{n}
\]
同样的，所有极限点的下确界为下极限，记作
\[
	\varliminf _{n\rightarrow \infty }a_{n}
\]
\smallskip

\no {\bf 定义2}

构造$\{b_m\}_{m\ge1}$，令$b_m = \sup\{a_m,a_{m+1},\cdots\}$，显然可知$\{b_m\}$单调递减，原数列的上极限可以定义为
\[
\begin{aligned}
	\varlimsup _{n\rightarrow \infty }a_{n} &= \lim_{m\to\infty} b_m = \inf\{b_m\}\\
	&= \lim_{m\to\infty}\left(\sup_{n\ge m}{\{a_n\}}\right)\\
	&= \inf_{m\ge1}\left(\sup_{n\ge m}{\{a_n\}}\right)\\
	&= \limsup _{n\rightarrow \infty }\{a_{n}\}
\end{aligned}
\]
类似的，我们可以构建$\{c_m\}_{m\ge1}$，令$c_m = \inf\{a_m,a_{m+1},\cdots\}$，显然可知$\{c_m\}$单调递增，原数列的下极限可以定义为
\[
\begin{aligned}
	\varliminf _{n\rightarrow \infty }a_{n} &= \lim_{m\to\infty} c_m = \sup\{c_m\}\\
	&= \lim_{m\to\infty}\left(\inf_{n\ge m}{\{a_n\}}\right)\\
	&= \sup_{m\ge1}\left(\inf_{n\ge m}{\{a_n\}}\right)\\
	&= \liminf _{n\rightarrow \infty }\{a_{n}\}
\end{aligned}
\]

\subsubsection{上、下极限的性质}

\no{\bf 性质1}

设$\{d_k\}$是$\{a_n\}$的收敛子序列，则有
\[
	\varliminf _{n\rightarrow \infty}a_{n} \le \lim_{k\to\infty}d_k \le \varlimsup _{n\rightarrow \infty}a_{n}
\]

\no{\bf 性质2}

数列$\{a_n\}$收敛的充要条件是，数列的上下极限相等，且为有限实数，即
\[
	\lim_{n\to\infty}a_n = L \iff \varliminf _{n\rightarrow \infty}a_{n} = \varlimsup _{n\rightarrow \infty}a_{n} = L \in \R
\]

\subsection{函数列的上、下极限}

对于在区间$I$上的函数列，上极限是一个在区间$I$上的函数，记作
\[
	\varlimsup_{n\to\infty} f_n(x)
\]
而其函数取值如下，其中$\left\{f_1(x),f_2(x),\cdots\right\}$是将自变量$x$带入函数列得到的数列
\[
	\varlimsup_{n\to\infty} f_n(x) = \varlimsup\left\{f_1(x),f_2(x),\cdots\right\}
\]
同理我们也可以得到如下的下极限
\[
	\varliminf_{n\to\infty} f_n(x) = \varliminf\left\{f_1(x),f_2(x),\cdots\right\}
\]
\smallskip

与数列极限的性质相似，我们可以得到
\[
	\varliminf_{n\to\infty} f_n(x) \le \varlimsup_{n\to\infty} f_n(x)
\]
\bigskip

\section{预备知识：集合论}

\subsection{集合基本概念}

\subsubsection{基本集合}

% 在课上的全集使用的是X 而空集是\Phi
我们将用$X$表示全集，$\Phi$表示空集

\subsubsection{集合簇}

我们称由集合组成的集合为集合簇，记作
\[
	\{A_\alpha\}_{\alpha\in I}
\]
其中$A_\alpha \subseteq X$，$I$是指标集（指标集不一定是$\mathbb{N}_+$也可以是$\R$，或其他任意集合）

\subsection{集合运算}

\subsubsection{基础运算}

交集记作$A\cap B$，并集记作$A\cup B$，补集记作$A^c$

\subsubsection{减法运算}

集合减法记作$A\setminus B = A\cap B^c$

\subsubsection{对称差}

对称差记作$A\triangle B = A\oplus B = (A\setminus B)\cup(B\setminus A)$

\subsubsection{大型运算}

\[
\begin{aligned}
	\bigcup_{n=1}^\infty\left(a+\frac1n,\infty\right)&=(a,+\infty)\\
	\bigcap_{n=1}^\infty\left(-\infty,a+\frac1n\right]&=(-\infty,a]\\
	\bigcap_{n=1}^\infty\left(a-\frac1n,\infty\right)&=[a,+\infty)\\
	\bigcup_{n=1}^\infty\left(-\infty,a-\frac1n\right]&=(-\infty,a)
\end{aligned}
\]

\subsection{集合列的上、下极限}

设$\{A_n\}_{n\ge1}$为集合列，其中$\forall A_n\subseteq X$。为定义集合列的上极限，我们构造集合列$\{B_m\}$如下：
\[
	B_m = \sup_{n\ge m}\{A_n\} = \bigcup_{n\ge m} A_n
\]
按包含关系，显然$B_1\ge B_2\ge \cdots$，可定义
\[
	\varlimsup_{n\to\infty}\{A_n\}=\lim_{m\ge 1}\{B_m\}=\bigcap_{m\ge1}B_m=\bigcap_{m\ge1}\bigcup_{n\ge m} A_n
\]
类似的，我们可以构造集合列$\{C_m\}$ 如下：
\[
C_m = \inf_{n\ge m}\{A_n\} = \bigcap_{n\ge m} A_n
\]
按包含关系，显然$C_1\le C_2\le \cdots$，可定义
\[
	\varliminf_{n\to\infty}\{A_n\}=\lim_{m\ge 1}\{C_m\}=\bigcup_{m\ge1}C_m=\bigcup_{m\ge1}\bigcap_{n\ge m} A_n
\]
我们也可以将上、下极限分别定义为
\[
	\varlimsup_{n\to\infty}\{A_n\}=\{x\in X: x\in \mbox{无穷多个}A_n\}
\]
\[
	\varliminf_{n\to\infty}\{A_n\}=\{x\in X: \exists\,N\in\mathbb{N} \st x\in A_n,\forall n\ge N\}
\]
这里我们只给出两种下极限定义等价性的证明（上极限的情况留给大家作思考题）: 
\[ 
    x\in\varliminf_{n\to\infty}\{A_n\}=\bigcup_{m\ge1}C_m\Leftrightarrow \exists\,N\in\mathbb{N} \st x\in C_N=\bigcap_{n\ge N} A_n \Leftrightarrow x\in A_n,\forall n\ge N
\]
显然
\[
	\varliminf_{n\to\infty}\{A_n\} \subseteq \varlimsup_{n\to\infty}\{A_n\}
\]
\subsection{映射}
设$X$、$Y$是两个集合，如果$X$中任一个元素$x$都按照某种规律$f$对应到$Y$的某一元素（记为$f(x)$），我们则称$f$为一从$X$到$Y$的映射，通常记为：
\[
	f:X\to Y
\]
对于$\forall A\subseteq X$，我们称$f(A)=\{f(x):x\in A\} $为$A$的像；对于$\forall B\subseteq Y$，我们称$f^{-1}(B)=\{x\in X:f(x)\in B\} $为$B$的原像
\subsubsection{单射}
满足以下条件的映射称为单射
\[
	x_1 \ne x_2 \Rightarrow f(x_1) \ne f(x_2)
\]
\subsubsection{满射}
满足以下条件的映射称为满射
\[
	f(X)=Y
\]
\subsubsection{双射}

既是单射又是满射的映射称为双射（或一一对应）
\subsubsection{逆映射}

当映射$f:X\rightarrow Y$是双射时，$f^{-1}$ 成为映射，称为 $f$ 的逆映射，记为
\[
	f^{-1}:Y\rightarrow X
\]

\subsubsection{映射与集合运算}

对于任何映射$f:X\rightarrow Y$，有集合列$\{A_\alpha\subseteq X\}_{\alpha\in I}$，$\{B_\alpha\subseteq X\}_{\alpha\in I}$，我们有
\[f\left(\bigcup_{\alpha\in I} A_\alpha\right) = \bigcup_{\alpha\in I}f(A_\alpha)\]
\[f\left(\bigcap_{\alpha\in I} A_\alpha\right) \subseteq \bigcap_{\alpha\in I}f(A_\alpha)\]
\[f^{-1}\left(\bigcup_{\alpha\in I} A_\alpha\right) = \bigcup_{\alpha\in I}f^{-1}(A_\alpha)\]
\[f^{-1}\left(\bigcap_{\alpha\in I} A_\alpha\right) = \bigcap_{\alpha\in I}f^{-1}(A_\alpha)\]
\[f^{-1}\left(B_\alpha^c\right) = \left(f^{-1}(B_\alpha)\right)^c\]


\subsection{集合的基数（势）}

粗略地讲，任一集合 $A$ 的基数或者势，记为 $|A|$，是指 $A$ 所含元素的个数。我们称自然数集
\[
	\mathbb{N}=\{1,2,\cdots\}
\]
的基数为可数无穷，记为 $\aleph_0$（读作：阿列夫0）。

在比较任意两个集合 $A$ 和 $B$ 的基数时，我们有下面的定义：

\no {\bf 定义1} 我们称 $\lvert A \rvert \le \lvert B \rvert$，如果存在从 $A$ 到 $B$ 的单射。\\
\no {\bf 定义2} 我们称 $\lvert A \rvert = \lvert B \rvert$，如果$\lvert A \rvert \le \lvert B \rvert$ 且 $\lvert B \rvert \le \lvert A \rvert$，也就是说，存在从 $A$ 到 $B$ 的双射。\\

\EXP 记 $\mathbb{N}_2=\{2,4,\cdots\}$，即所有偶自然数的集合。不难看出 $\lvert\mathbb{N}_2\rvert=\lvert\mathbb{N}\rvert=\aleph_0$，因为每个自然数乘 $2$ 就是一从 $\mathbb{N}$ 到 $\mathbb{N}_2$ 的双射。

\subsubsection{幂集}
称任一集合$A$的所有子集构成的集合为集合$A$的幂集，记为$P(A)$。不难看出
\[
\lvert P(A) \rvert = 2^{\lvert A \rvert}
\]

\subsubsection{无最大基数定理}

对于任一非空集合 $A$，我们总有 $\lvert A \rvert<\lvert P(A) \rvert$。
\smallskip

\pro

我们先证明 $\lvert A \rvert \le \lvert P(A) \rvert$：\\
定义映射 $f:A\rightarrow P(A)$ 如下：
\[ 
	g(x)=\{x\}, \forall x \in A
\] 
不难看出，$g$ 是单射。

下面我们用反证法证明 $\lvert A \rvert \ne \lvert P(A) \rvert$:
若相等，则存在双射
\[
	f:A\rightarrow P(A)
\]
定义一个$A$的子集合$B$如下：
\[
B = \{x\in A: x\notin f(x)\}
\]
由于 $f$ 是双射，所以必存在（唯一）元素 $y\in A$ 使得
\[
f(y) = B
\]
考虑 $y$ 与 $B$ 的关系，发现无论如何都会产生矛盾：
\ITEMS
	\item 如果 $y\in B=f(y)$, 由 $B$ 的定义，我们会得到 $y\notin B$；
	\item 如果 $y\notin B=f(y)$, 由 $B$ 的定义，我们会得到 $y\in B$。
\ITEME

\no {\bf 注} 我们记 $\aleph_n=2^{\aleph_{n-1}}, n=1, 2, \cdots$。由上述定理，显然我们有 
\[
	\aleph_0 < \aleph_1 < \aleph_2 <\cdots< 
\]

\subsubsection{关于基数的几个基本定理}
\ITEMS
	\item 对任意集合 $A$ 和 $B$，若 $A\subseteq B$，则 $\lvert A \rvert \le \lvert B \rvert$。
	\item 设 $A$ 是可数集（有限集或可数无穷集），$B$ 是无穷集，则 $\lvert A\cup B \rvert= \lvert B \rvert$。
     （例：Hilbert旅馆）
	\item 设集合列 $\{A_n\}_{n=1}^{\infty}$ 中每一 $A_n$ 的基数都是 $\aleph_0$，则 $\lvert \bigcup_{n=1}^\infty A_n \rvert=\aleph_0 $。也就是说，可数个可数集的并集还是可数集。 （例：记 $\mathbb{Q}=$ 有理数集，则 $\lvert\mathbb{Q}\rvert=\aleph_0 $） 
\ITEME


\subsubsection{实数集的基数}
实数集 $\R$ 的基数一般记为 $c$。 证明 $c=\aleph_1$。\\
\pro

我们先证明 $\lvert \R \rvert=\lvert (0,1]\rvert $，分三步进行：\\

1. $\lvert \R\rvert =\lvert (-1,1)\rvert $: 不难看出映射  $f:(-1,1)\rightarrow \R$ 满足 $f(x)=\frac {x}{1-x^2}$ 是双射。\\

2. $\lvert (-1,1)\rvert=\lvert (0,1)\rvert $: 不难看出映射  $f:(-1,1)\rightarrow (0,1) $ 满足 $f(x)=\frac {x+1}{2}$ 是双射。\\

3. $\lvert (0,1)\rvert=\lvert (0,1]\rvert $： 因为 $(0,1)\subseteq (0,1]\subseteq (-1,1)$。\\

下面我们证明 $\lvert (0,1]\rvert=2^{\aleph_0}$: \\
1. 把 $(0,1]$ 中的任一实数 $x$ 用二进制表示：更具体地讲，就是找到每项取值是 $0$ 或 $1$ 的无穷数列 $\{a_n\}_{n \ge 1}$ 满足在二进制下 $x=0.a_1 a_2\cdots a_n \cdots$，其中无穷多个 $a_n$ 是 $1$ 。\\
\ITEMS 
	\item 第一步: 把区间 $(0,1]$ 二等分为两半开闭子区间：若 $x$ 属于靠左的（称为第一个，后同）子区间 $(0,\frac{1}{2}]$，则 $a_1=0$；若 $x$ 属于靠右的（称为第二个，后同）子区间，则 $a_1=1$；
	
	\item 第二步：把 $x$ 所属的第一步中的小区间二等分为两半开闭子区间:  若 $x$ 属于第一个子区间，则 $a_2=0$；若 $x$ 属于第二个子区间，则 $a_2=1$；
	\item $\vdots$ 
	\item 第 $n$ 步： 把 $x$ 所属的上一步中的小区间二等分为两半开闭子区间:  若 $x$ 属于第一个子区间，则 $a_n=0$；若 $x$ 属于第二个子区间，则 $a_n=1$；
	\item $\vdots$
\ITEME
    注意到上述 $a_n$ 中，有无穷多个是 $1$ （为什么？）。这样我们就建立了 $(0,1]$ 和集合 $$S=\{(a_1, a_2,\cdots, a_n, \cdots): \mbox{每个 $a_n$ 是 $0$ 或 $1$ 且 其中无穷多个 $a_n$ 是 $1$} \}$$ 的一一对应关系。\\
2 决定 $\lvert S\rvert$：令 $$T=\{(a_1, a_2,\cdots, a_n, \cdots): \mbox{每个 $a_n$ 是 $0$ 或 $1$ 且 其中只有限多个 $a_n$ 是 $1$} \}$$ 和 $$ R=\{(a_1, a_2,\cdots, a_n, \cdots): \mbox{每个 $a_n$ 是 $0$ 或 $1$}\}$$ 
不难看出 $\lvert R\rvert=2^{\aleph_0}, \lvert T\rvert=\aleph_0$ （为什么？）且 $S\cup T= R$。由上面关于基数的基本定理 $2$，我们有 $$\lvert S\rvert=\lvert S\cup T\rvert =\lvert R\rvert=2^{\aleph_0}=\aleph_1$$

\subsubsection{连续统假设}
 
 \no {\bf 连续统假设} 不存在一个无穷集合，其基数严格介于自然数集的基数 $\aleph_0$ 与实数集的基数 $\aleph_1$（即连续统的基数）之间。 \\
 \no {\bf 广义连续统假设} 不存在一个无穷集合，其基数严格介于 $\aleph_n$ 与 $\aleph_{n+1}, n=0, 1,2, \cdots $ 之间。

在数学界广泛采用的ZFC集合公理系统下：
\ITEMS
\item Godel 证明了该假设的相容性：即不能证明它是错的； 
\item Cohen 证明了它的独立性：即不能证明它是对的。
\ITEME

\section{预备知识：$\R^n$上的拓扑}
这一部分我们研究 $n$ 维欧氏空间 $\R^n$ 上的拓扑（Topology 地形）
% 在R^n上类比一维中用绝对值定义距离来用范（绝对值）定义距离	

\subsection{开集与闭集}

\subsubsection{开球、邻域}
\ITEMS
	\item 我们把 $\R^n$ 上的开球定义为 % 这里定义开球所用的距离（模）是否需要提前定义，映射d
\[
	B(x, r)=\{y\in\R^n:\lvert y-x\rvert<r\}
\]
也就是以 $x$ 为圆心，半径小于 $r$ 的所有点的集合。
	\item 集合 $U\subseteq \R^n$ 称为 $x$ 的邻域如果存在开球 $B(x, r)\subseteq U$。
\ITEME

具体而言，对于 $\R^n$ 中任一点 $x$ 附近的地形，我们是用 $x$ 的邻域，本质上是用半径 $r$ 很小的开球 $B(x, r)$来刻画的。

\subsubsection{开集与闭集的第 1 种定义}

\ITEMS
	\item 集合$A\subseteq\R^n$称为开集，如果它是若干开球的并集，即
\[
	A = \bigcup_{\alpha\in I} B_\alpha(x_\alpha,r_\alpha)
\]

	\item 集合 $A\subseteq\R^n$ 称为闭集，如果其补集 $A^c$ 是开集，即
\[
	A = \left[\bigcup_{\alpha\in I} B_\alpha(x_\alpha,r_\alpha)\right]^c
\]
\ITEME

\no {\bf 注1} 我们在 $\R^n$ 中引入开集是为了刻画更大范围的地形，自然而然地，我们会想到把其中每一点附近的地形拼起来，于是自然就有了上述开集的定义。\\

\no {\bf 注2} 更一般地，为了定义一非空集合 $X$ 上的拓扑，我们只需定义 $X$ 中的所有开集（或等价地，所有闭集），通常多用开集来定义。自然而然地，我们称这样配备了拓扑的非空集合为拓扑空间。\\

\no {\bf 注3} 设 $B$ 是由拓扑空间 $X$ 上的若干开集构成的集合簇，如果 $X$ 中的任一开集都是 $B$ 中成员的并（可以任意多个），我们则称 $B$ 是为拓扑空间 $X$ 的一拓扑基。显然，$\R^n$ 中所有开球构成的集合是 $\R^n$ 的一拓扑基。\\	% 可以冗余

\no 为了更好地理解开集与闭集，下面我们将讨论它们的第二种定义，为此我们先引入以下定义：\\

对任一集合 $A \subseteq \R^n$  

\ITEMS
	\item {\bf 内点} 点 $x\in A$ 称为 $A$ 的内点，如果存在以 $x$ 为球心的开球 $B(x;r)$ 使得该开球包含于 $A$，即
	\[
		\exists r > 0 \st B(x;r)\subseteq A
	\]
	我们把 $A$ 的所有内点构成的集合称为 $A$ 的内部，记为 $\mathring{A}$。
	
	\item {\bf 边界点} 点 $x\in \R^n$ 称为 $A$ 的边界点，如果 $x$ 既不是 $A$ 的内点也不是 $A^c$ 的内点，即
	\[
		\forall r > 0 \st B(x;r)\mbox{中既有}A\mbox{中的点，也有}A^c\mbox{中的点}
	\]
	我们把 $A$ 的所有边界点构成的集合称为 $A$ 的边界，记为 $\partial{A}$。

	\item {\bf 聚点} 点 $x\in \R^n$ 称为 $A$ 的聚点（极限点），如果任意以 $x$ 为球心的开球都含 $A$ 内的非 $x$ 点，即	% 这一部分是根据课上第二种定义写的
	\[
		\forall r>0 ,\ \exists a\in B(x;r)\st (a\in A)\land(a\neq x)
	\]
	也可以等价定义为
	\[
		\exists\{x_n\}_{n=1}^\infty,\ \forall x_n\in A \st\lim_{n\to\infty}x_n=x\\
	\]
	我们把 $A$ 的所有聚点构成的集合称为 $A$ 的导集，记为 $A'$。

	\item {\bf 孤立点} 点 $x\in A$ 称为 $A$ 的孤立点，如果存在以 $x$ 为球心的开球，仅与集合 $A$ 交于点 $x$，即
	\[
		\exists r>0 \st B(x;r)\cap A=\{x\}
	\]
	
	\item {\bf 闭包} 集合 $\overline A\subseteq \R^n$  称为 $A$ 的闭包，如果{\bf 定义1：} $\overline A = A \cup A'$ 或者{\bf 定义2：} $\overline A = \mathring {A} \cup \partial {A}$ 。\\

{\bf 注} 闭包的定义 $1$ 是从分析的角度来考虑问题的，就是希望任一点集在加入其所有的极限点后，所得集合在极限运算下封闭：即其内任何柯西序列的极限点必含于其中（事实确实如此，参见下面的例 $2$ ）。而闭包的定义 $2$，讲的则是从几何直观上，如何把任一点集给封闭起来。\\

下面我们说说闭包这两种定义的等价性：
\ITEMS
	\item 一方面，对任意 $x \in A \cup A'$：若 $x \notin \mathring {A}$，由内点定义知，$x$ 的任一开球 $B(x; r)$ 必含 $A^c$ 中点，此时
		\ITEMS
			\item 若 $x \in A$，则 $B(x; r)$ 同时还含 $A$ 中点 $x$；
			\item 若 $x \in A'$，则由聚点定义，$B(x; r)$ 同时必含 $A$ 中点。
		\ITEME
	这也就是说 $x$ 是边界点，即 $x \in \partial {A}$，从而 $A \cup A' \subseteq \mathring {A} \cup \partial {A}$
	\item 另一方面，由边界点和聚点的定义不难看出：对任意非空集合 $A$，不在 $A$ 中的 $A$ 的边界点必为 $A$ 的聚点。从而 
	$\mathring {A} \cup \partial {A} \subseteq A \cup A'$。
	
\ITEME

\no {\bf 例 1} 在 $\R$ 中，$(0, 1]$ 的闭包 是 $[0, 1]$。\\

\no {\bf 例 2} 证明：对 $\R^n$ 中任一子集 $A$，其闭包 $\overline A$ 的闭包仍是 $\overline A$，即 $\overline {\overline A} = \overline A$。该证明我们不作要求，大家可以试一下。\\


\ITEME

\subsubsection { 开集与闭集的第 2 种定义}

\ITEMS

	\item 集合 $A\subseteq\R^n$ 称为开集，如果 $A$ 的所有点都是内点，即 $A = \mathring A$。\\

	\item 集合 $A\subseteq\R^n$ 称为闭集，如果 $A = \overline A$，也就是说 $A$ 的所有聚点都含于 $A$，即 $A' \subseteq A$。特别地，如果 $A'= A$，我们则称 $A$ 为完全集。\\

\ITEME

\bigskip

\pro 

	两种开集定义的等价性是比较明显的。下面我们给出的两种闭集定义等价性的证明：
	\[
		\begin{aligned}
			A\mbox{是闭集}\Leftrightarrow A = \overline A 
			&\Leftrightarrow A = A \cup A'\\ % 闭包的定义
			& \Leftrightarrow A' \subseteq A\\
			& \Leftrightarrow A^c \cap A' = \Phi % \varnothing
		\end{aligned}
	\]
	也就是说， $A^c$ 中任一点必不是 $A$ 的聚点，这等价于
	\[
		\begin{aligned}
			&\quad \forall x\in A^c,\ \exists r>0 \st B(x,r)\cap A = \Phi\\
			&\Leftrightarrow \forall x\in A^c,\ \exists r>0 \st B(x,r)\subseteq A^c\\ % 根据内点的定义
			&\Leftrightarrow A^c \mbox{的任一点都是的内点} \\
			&\Leftrightarrow A^c \mbox{是开集}\\
			&\Leftrightarrow A \mbox{是闭集}
			% &\Rightarrow A^c = \mathring {A^c}
			% 本来想写 x\in \mathring{A^c} 但太不明确了
		\end{aligned}
	\]
\no {\bf 注1} 由上述定义易知：全集 $\R^n$ 既是开集又是闭集，从而其补集空集 $\Phi$ 也既是开集又是闭集。一般地，任一拓扑空间中全集和空集都既是开集又是闭集。\\

\no {\bf 注2} 由上述 {\bf 例 2} 知，$\R^n$ 中任一子集 $A$ 的闭包 $\overline A$ 是闭集。 事实上，$\overline A$ 还是包含 $A$ 的最小闭集：设 $B$ 是任一包含 $A$ 的闭集，显然 $ A' \subseteq B'$，从而 $\overline A = A \cup A'\subseteq B \cup B'= \overline B = B$。\\

\no {\bf 注3} 由上述定义易知：闭集就是对极限运算封闭的点集，也就是说：对该集合内任一柯西序列，其极限点必在该集合内。\\

\subsection{连续函数的等价定义}

考虑函数 $f:X\to\R$，其中 $X\subseteq\R^n$。先回忆下我们在数学分析中是如何用 $\varepsilon-\delta$ 语言来定义 $f$ 的连续性：

我们称函数 $f$ 在 $X$ 上连续，如果 $f$ 在 $X$ 的每一点上连续。也就是说： $\forall a\in X$ 我们都有
\[
	\forall \varepsilon > 0,\ \exists \delta > 0\ \mbox{使得}\ \forall x\in X \st \ \lvert x - a\rvert <\delta \Rightarrow \lvert f(x) -f(a)\rvert<\varepsilon
\]
上式显然等价于
\[
	\forall a \in X,\ \forall \varepsilon > 0,\ \exists \delta > 0\ \mbox{使得}\ \forall x\in X \st \ x\in B(a;\delta)\Rightarrow f(B(a;\delta)) \subseteq B(f(a);\varepsilon)
\]
这就是说：$f(x)$ 在 $x=a$ 处连续如果 $f(a)$ 的任一邻域在 $f$ 下的原像是 $a$ 的邻域。

\bigskip

下面我们给出连续函数 $f$ 的三种等价定义

\ITEMS
	\item $f$ 连续的 $\varepsilon-\delta$ 定义（见上）
    \item $f$ 连续如果 $\R$ 中任一开集在 $f$ 下的原像仍是开集
	\item $f$ 连续如果 $\R$ 中任一闭集在 $f$ 下的原像仍是闭集
\ITEME

\pro

注意到 $f^{-1}(A^c)=(f^{-1}(A))^c$，定义2、3 的等价性是比较明显的。\\

定义2 $\Rightarrow$ 定义1: 对 $a \in X$ 及 $\varepsilon > 0$，$\R$ 中开集 $B(f(a);\varepsilon)$ 在 $f$ 下的原像 $f^{-1}\left[B(f(a);\varepsilon)\right]$ 是 $X$ 中的开集，从而 $a$ 是其内点，也就是说：
\[
	\exists \delta > 0 \st B(a;\delta)\subseteq f^{-1}\left[B(f(a);\varepsilon)\right]
\]
这显然等价于 $f(B(a;\delta)) \subseteq B(f(a);\varepsilon)$。\\

\subsection{$\sigma$-代数}
% sigma 代表可数无穷求和
\subsubsection{$\sigma$-代数}

对于由全集 $X$ 的若干子集合构成的集合簇 $S$， 我们称 $S$ 为 $X$ 上的一 $\sigma$-代数，如果它满足下述条件：
\ITEMS
	\item 全集 $X \in S$；
	\item $S$ 对补集运算封闭，即 $\forall A\in S \Rightarrow A^c\in S$
	\item $S$ 对可数多个集合的并集运算封闭 ，即
	\[
		\forall A_1,A_2,\cdots,A_n,\cdots\in S\ ,\ \bigcap_{n=1}^\infty A_i\in S
	\]

\no {\bf 注1} 显然空集 $\emptyset \in S$。\\	
\no {\bf 注2} 根据 DeMorgan 定律和上述条件 2，条件 3 与下述条件等价：\\
$S$ 对可数多个集合的并集运算封闭 ，即
	\[
		\forall A_1,A_2,\cdots,A_n,\cdots\in S\ ,\ \bigcup_{n=1}^\infty A_i\in S
	\]
\ITEME

\subsubsection{波莱尔（Borel） $\sigma$-代数}

我们把 $\R^n$ 上含有 $\R^n$ 的所有开集的最小 $\sigma$ -代数（按包含关系）称为 Borel $\sigma$-代数，记作 $\mathcal{B}(\R^n)$ 。关于 Borel $\sigma$-代数，我们有下述性质：
\ITEMS 
	\item $\mathcal{B}(\R^n)$ 中含 $\R^n$ 中所有开集
	\item $\mathcal{B}(\R^n)$ 中含 $\R^n$ 中所有闭集
	\item $\mathcal{B}(\R^n)$ 中含 $\R^n$ 中所有 \(G_\delta\) 集，其中我们把可数无穷多个开集的交集称为 \(G_\delta\) 集 
	\item $\mathcal{B}(\R^n)$ 中含 $\R^n$ 中所有 \(F_\sigma\) 集，其中我们把可数无穷多个闭集的并集称为 \(F_\sigma\) 集
	\item[\vdots] \ 其他性质不做展开
\ITEME

\subsubsection{距离}

对于非空集合 $X$ ，存在以下映射
\[
	d: X \times X \to \mathbb{R}
\]
其中 $\times$ 为直积（笛卡尔积），如果映射 $d$ 满足以下条件
\ITEMS
	% 有的说法把正定性分为非负性和非退化性
	\item {\bf 正定性}\ $d(x, y) \ge 0$，且 $d(x, y) = 0 \iff x = y$
	\item {\bf 对称性}\ $\forall x,y\in X , d(x,y) = d(y,x)$
	\item {\bf 三角不等式}\ $d(x,y)+d(y,z) \ge d(z,x)$
\ITEME
我们称 $d$ 为 $X$ 上的一个距离（度量），称 $(X,d)$ 为一距离空间（度量空间）
% 在距离空间中可以使用开球，因为开球的定义中要使用映射 $d$
% \[
% 	 B(x;r)=\{y\in\R^n:\ d(y,x)<r\} 
% \]
\bigskip

\EXP 设 $X = C[a,b]$ 是 从 $[a,b]$ 到 $\R$ 上的所有连续函数构成的集合。对于 $\forall f, g \in X$，我们定义
\[
	d(f,g) = \max_{a\le x\le b}\lvert f(x)-g(x)\rvert
\]
不难验证 $(X,d)$ 是一距离空间。

\subsection{拓扑空间}

\subsubsection{拓扑空间的定义}

对于拓扑空间 $X$ ，以及 $X$ 的一子集簇 $\tau$，我们称 $(X,\tau)$ 为拓扑空间 ($\tau$ 为其开拓扑，$\tau$ 中子集为开集)，如果 $\tau$ 满足下述条件：
\ITEMS
\item $\varnothing , X \in \tau$
\item 任意并封闭
\[
\forall \{ A_\alpha\}_{\alpha \in I} \subseteq \tau \Rightarrow \bigcup_{\alpha \in I} A_\alpha \in \tau
\]
\item 有限交封闭
\[
\forall A , B \in \tau \Rightarrow A \cap B \in \tau
\]
\ITEME

{\bf 注1} 相应地，对于非空集合 $X$，我们也可以通过定义闭拓扑（即 $X$ 中所有闭集构成的集合） 来定义拓扑空间：只需把上面的 “任意并封闭” 改为 “任意交封闭”，“有限交封闭” 改为 “有限并封闭” 即可。\\

{\bf 注2} 距离空间亦称度量空间是特殊的拓扑空间：只需类比 $\R^{n}$ 来定义其开球，再把开集定义为开球的任意并即可。

\subsubsection{覆盖}

设 $S$ 是拓扑空间 $X$ 的子集，我们称 $X$ 的子集簇 $\mathcal{A}=\{A_i\}_{i \in I}$ 是 $S$  的一个覆盖如果 
\[
S\subseteq \bigcup_{i \in I} A_i
\]
\smallskip

特别地，若 $\mathcal{A}$ 中的每一元素都是开集，则称 $\mathcal{A}$ 是 $S$ 的一个开覆盖。
\smallskip

若存在 $\mathcal{A}$ 的子集 $\mathcal{B}$ 仍是 $S$ 的覆盖，则称 $\mathcal{B}$ 是 $\mathcal{A}$ 的一个子覆盖；若 $\mathcal{B}$ 是有限集，则称之为有限子覆盖。

\subsubsection{紧集}

设 $A$ 是拓扑空间 $X$ 的子集，我们称 $A$ 为紧集如果 $A$ 的任一开覆盖都有有限子覆盖。

在距离空间中，紧集一定为有界闭集：$\text{紧集} \,\Rightarrow\,\text{有界闭集}$

特别地，在 $\R^n$ 中紧集等价于有界闭集：$\text{紧集} \,\iff\,\text{有界闭集}$

\subsection{赋范空间}

对于线性空间 $X$ ，如果能建立映射 $\lVert\cdot\rVert: X \to \mathbb{R}_{\ge 0}$，使其满足如下条件
\ITEMS
\item 正定性 \[\lVert x\rVert = 0 \iff x =0 \]
\item 齐次性 \[\quad\lVert r x\rVert = |r| \lVert x\rVert, \; \forall x \in X\]
\item 三角不定式 \[\lVert x + y\rVert \le \lVert x\rVert + \lVert y\rVert, \;\forall x, y \in X\]
\ITEME
我们则称 $(X,\lVert\cdot\rVert)$ 为赋范空间，$\lVert\cdot\rVert$ 为 $X$ 上的范（norm）。对任意 $x \in X, \lVert x\rVert$ 称为 $x$ 的范数。\\

{\bf 注1} 事实上，范就是类比于 $\R^n$ 中的向量长度（ 即 $\R^1$ 中的绝对值的推广） 而来的。当然类似于 $\R^n$，我们也可以自然而然地在任一赋范空间中定义距离，从而使其成为距离空间。\\

{\bf 注2} 我们称完备（取极限封闭）的赋范空间为巴拿赫空间（Banach）。

\bigskip

\EXP 设 $X = C[a,b]$ 是 从 $[a,b]$ 到 $\R$ 上的所有连续函数构成的集合。对于 $\forall f \in X$，定义 
\[
\lVert f \rVert = \max_{a\le x\le b}\lvert f(x)\rvert
\]
不难验证，如上定义的 $\lVert \cdot \rVert$ 是 $X$ 上的范。\\

自然地，对任意 $f, g \in X$，我们定义其距离
\[
d(f,g) = \lVert f-g \rVert
\]
这样，就得到了我们在以前的例中所讨论的距离空间 $(X,d)$。

对任意 $x \in \mathcal{C}$，必有 $x\in \mbox{某个}F_{mn}, m=1, 2, \cdots$。注意到该闭区间 $F_{mn}$ 的长度为 $\frac{1}{3^m}$ 且 $\lim\limits_{m\to\infty}\frac{1}{3^m}=0$，因此对任意 $\epsilon > 0$，在 $m$ 足够大时，开球 $B(x,\epsilon)$ 必包含整个 $F_{mn}$，从而 $B(x,\epsilon)\cap \mathcal{C}$ 必含 $F_{mn}$ 的端点，这表明任意 $x \in \mathcal{C}$ 都是 $\mathcal{C}$ 的聚点，所以 $\mathcal{C}$ 是完全集。

\subsubsection{康托三分集的基数}

不难看出，康托三分集 $\mathcal{C}$ 中每一个数与三进制小数集 $\{0.a_1a_2a_3\cdots a_n\cdots: \mbox{每个 $a_n$ 是 $0$ 或 $2$}$\} 一一对应，所以
\[
\lvert \mathcal{C}\rvert = 2^{\aleph_0}
\]

\subsubsection{康托三分集的“长度”}

这里，康托三分集 $\mathcal{C}$ 的 “长度” 理解为总长度 1 减去所有移除的开区间 $I_{mn}$ 的长度。注意到第 $m$ 次移除 $2^{m-1}$ 开区间，每个长度均为 $\frac{1}{3^m}$，从而康托三分集 $\mathcal{C}$ 的 “长度” 等于
$$ 1-\sum_{m=1}^{\infty}\frac {2^{m-1}}{3^{m}}=1-\frac{\frac{1}{3}}{1-\frac{2}{3}}=0 $$

\subsubsection{$\varphi$ 映射}

建立以下映射 $\varphi:\mathcal{C} \to [0,1]$ ，把康托集的三进制小数映射到二进制，将所有 $2$ 替换为 $1$
\[
x =\ 0.a_1 a_2 a_3 \cdots\ \mapsto\ \varphi(x) =\ 0.b_1 b_2 b_3\cdots \quad \mbox{其中}\ b_i = \frac{a_i}{2}, \forall i \in \mathbb{N}
\]
显然可知

\ITEMS
\item $\varphi$ 是满射
\item $\varphi$ 非严格单调增（不减）

\EXP $\varphi(\frac13) = \varphi(\frac23)=\frac12$

\item $\varphi$ 连续
\ITEME

\subsubsection{康托函数 $\varPhi(x)$}

建立以下映射 $\varPhi:[0,1] \to [0,1]$，有以下对应关系
\[
\varPhi(x) = \begin{cases}
	\varphi(x) &,x\in\mathcal{C}\\
	\varphi(I_{mn}\mbox{的左端点})= \varphi(I_{mn}\mbox{的右端点}) &, x\in I_{mn}
\end{cases}
\]
不难看出

\ITEMS
\item $\varPhi$ 是满射
\item $\varPhi$ 非严格单调增（不减）
\item $\varPhi$ 连续
\item 虽然 $\varPhi$ 在 $[0, 1] $ 上不是常数函数，但由于它在每一 $I_{mn}$ 上都是常数（因 $I_{mn}$ 不同而不同），所以其导数在 $[0, 1] $ 上几乎处处为 $0$
\[
\{ x\in [0,1] : \varPhi'(x) = 0 \} \supseteq \bigcup_{m=1}^\infty\bigcup_{n=1}^{2^{n-1}}I_{mn}
\] 
$\bigcup\limits_{m=1}^\infty\bigcup\limits_{n=1}^{2^{n-1}}I_{mn}$ 的长度为 $\sum\limits_{n=1}^{\infty}\frac {2^{n-1}}{3^{n}} = 1$
\[
\{ x\in [0,1] : \varPhi'(x) \neq 0 \} \subseteq \mathcal{C}
\] 
$\mathcal{C}$ 的长度为 $0$
\ITEME

\EXP 证明（参见 HW1）：对任意 $x\in [0,1]$，都有 $\varPhi(x) + \varPhi(1-x)=1$。

\subsubsection{$\varPsi$ 映射}

在康托函数的基础上，我们构建一严格单调增的函数 $\varPsi:[0,1] \to [0,1]$
\[
\varPsi(x) =  \frac{x+\varPhi(x)}{2}
\]
不难看出，$\varPsi$ 连续、双射。

\subsubsection{康托函数在 $[0, 1]$ 上的黎曼积分}

注意到 $\varPhi(x) + \varPhi(1-x)=1$，对此式两边分别在 $[0, 1]$ 上取黎曼积分，得到
$$\int_{0}^{1}\varPhi(x)\mathrm{d} x + \int_{0}^{1}\varPhi(1-x)\mathrm{d} x=1$$
由区间重现定理，$\int_{0}^{1}\varPhi(x)\mathrm{d} x=\int_{0}^{1}\varPhi(1-x)\mathrm{d} x$，从而黎曼积分
$$\int_{0}^{1}\varPhi(x)\mathrm{d} x=\frac12 $$

\subsubsection{康托函数在 $[0, 1]$ 上的勒贝格积分}

\chapter{勒贝格测度}

关于黎曼积分和勒贝格积分，勒贝格曾有一个形象的比喻：数一沓钱，黎曼积分是一张张的数，总钱数就是每一张的面值相加；而勒贝格积分则是先按面值分类，再分别去数每一面值的张数，然后再把每一面值乘上其相应的张数，最后再加起来就是总钱数。也就是说，我们所说的勒贝格积分就是先对积分函数（为便于研究，不妨先考虑非负函数，搞清楚后再一般化）按函数值，也就是所谓的 “高度” 进行分段，然后再测量每段对应的 “底” 的大小，最后再分别相乘再相加，得到所谓的总 “面积”。因此，我们首先希望能在 $\R^n$ 中引入一 “测度”，使得它是我们所熟悉的长度、面积、体积概念的推广，最好能测量 $\R^n$ 中任一子集的大小。也就是说，我们希望建立定义在 $\R^n$ 的幂集 $\mathcal {P} (\R^n) $ 上的广义非负实值函数 $\mathcal {M}$
\[
\mathcal {M}:\mathcal {P} (\R^n)\mapsto \R_{\ge0} \cup {+\infty}
\]
满足以下条件:
\ITEMS
	\item 非负性:\quad 对于$\forall A \subseteq \R^n$ 都有 $\mathcal {M} (A) \ge 0$ 且 $\mathcal {M}(\varnothing) = 0$
	\item 平移不变性:\quad 对于 $A,B\subseteq \R^n$ 且 $A$ 平移得 $B$ ，那么 $\mathcal {M}(A) = \mathcal {M}(B)$
	\item 规范性:\quad $\mathcal {M}([0,1]^n)=1$
	\item 可数可加性（$\sigma$-可加性）:\quad 
\[
\forall \{E_i\}_{i=1}^\infty \subseteq P(\R^n), \forall i\neq j,E_i\cap E_j =\varnothing \ \Rightarrow\ \mathcal {M}(\bigcup_{i=1}^\infty E_i) = \sum_{i=1}^\infty \mathcal {M}(E_i) 
\]
\ITEME
{\bf 注} 如果把上述条件 $4$ 中 $\{E_i\}$ 的个数换成有限个，则称之为有限可加性。事实上，人们认识到：相较于有限可加性，可数可加性要有用得多。可数可加性也正是我们本章要讲的勒贝格测度的核心特征。大家不妨回顾一下，在上一章计算康托集的 “长度” 时，实际上是在计算所有去掉的开区间 $I_{mn}$ 的总长度时，我们事实上已经用到了 “长度” 的可数可加性。

\section{勒贝格外测度}

\subsection{开矩体}

\subsubsection{开矩体的定义}
定义在 $\R^n$ 中的开矩体为
\[
I = (a_1,b_1)\times(a_2,b_2)\times\cdots\times(a_n,b_n)
\]
开矩体 $I$ 的闭包为闭矩体 
\[
\overline I = [a_1,b_1]\times[a_2,b_2]\times\cdots\times[a_n,b_n]
\]
定义如下形式为半开闭矩体
\[
\begin{aligned}
	(a_1,b_1]\times(a_2,b_2]\times\cdots\times(a_n,b_n]\\
	[a_1,b_1)\times[a_2,b_2)\times\cdots\times[a_n,b_n)	
\end{aligned}
\]
开矩体的体积为 $\lvert I\rvert = \prod\limits_{i=1}^n (b_i-a_i)$

\subsubsection{L-覆盖（可数开矩体覆盖）}

设 $A\subseteq \R^n$。若 $\{I_k\}_{k\ge1}$ 是 $\R^n$ 中的可数个开矩体 使得
\[
\bigcup_{k\ge 1} I_k \supseteq A
\]
则称 $\{I_k\}_{k\ge1}$ 为 $A$ 的一个 $L$-覆盖。

\subsection{勒贝格外测度}

下面我们来定义勒贝格外测度 $m^*: \mathcal {P}(\R^n) \to \R_{\ge0} \cup \{+\infty\}$。\\

对于 $\forall A\subseteq \R^n$，其勒贝格外测度 $m^*(A)$ 定义为
\[
m^*(A) = \inf\{\sum_{k\ge 1} \lvert I_k \rvert: \mbox {$\{I_k\}_{k\ge1}$ 为 $A$ 的
	 $L$-覆盖}\}
\]
也就是说，先计算 $A$ 的任一 $L$-覆盖 $\{I_k\}_{k\ge1}$ 的体积 $\sum_{k\ge 1} \lvert I_k \rvert$ （可以是 $+\infty$），然后所有这些体积的下确界，就是 $A$ 的勒贝格外测度 $m^*(A)$。

\ITEMS
\item 对于$\R^n$ 中任意有界集合，其外测度必有限
\item 当一个集合 $A$ 的外测度为 $+\infty$ ，当且仅当 $A$ 的任意可数开矩体覆盖，都有 $\sum\limits_{k\ge1}\lvert I_k \rvert = +\infty$
\ITEME

\EXP
\ITEMS
\item 独点集的外测度等于 $0$
\item 低维子集的外测度等于 $0$
\[
A = \left\lbrace\begin{bmatrix} c\\ x_2 \\ \vdots\\ x_n\end{bmatrix}: x_2,\cdots,x_n\in\R\right\rbrace \subseteq \R^n \Rightarrow m^*(A) = 0
\]
\item 开矩体的外测度等于其体积，也等于其闭包的外测度
\[
m^*(I) = \lvert I\rvert = m^*(\overline I)
\]
\pro 这里只给出 $m^*(I) = \lvert I\rvert$ 的证明。
\ITEMS
	\item 显然 $\lvert I\rvert \ge m^*(I)$ 成立
	\item 对于 $\forall \varepsilon > 0$ ，存在 $I$ 的可数开矩体覆盖 $\{I_k\}_{k\ge 1}$ 使得
\[
\lvert I \rvert \le \sum_{k\ge1} \lvert I_k \rvert \le m^*(I) +\varepsilon
\]
对 $\varepsilon$ 取极限，使其趋近于 $0$
\[
\lvert I \rvert \le m^*(I) 
\]
\ITEME
\ITEME

\subsubsection{外测度的性质}

\ITEMS
\item 非负性 \quad 对于 $\forall A \subseteq \R^n$ 都有 $m^*(A) \ge 0$ 且 $m^*(\varnothing) = 0$
\item 单调性 \quad 对于 $\forall A \subseteq B \subseteq \R^n$ 有 $m^*(A) \le m^*(B)$
\item 次可加性 \quad 对于 $\forall \{A_i\subseteq \R^n\}_{i\ge 1}$ 都有 
\[
m^*\left(\bigcup\limits_{i\ge 1} A_i\right) \le  \sum_{i \ge 1} m^*(A_i)
\]
注：即使 $A_i \cap A_j = \varnothing, \forall i \neq j$ 仍有可能取小于号

\pro 1, 2 显然，下面给出次可加性的证明。
\ITEMS
	\item 若 $\sum\limits_{k ge1} m^*(A_i)=+\infty$，显然次可加性成立。
	\item 设 $\sum\limits_{k ge1} m^*(A_i)$ 有限，则对于 $\forall i$，$m^*(A_i)$ 都有限。从而对 $\forall \varepsilon > 0$ ，$\exists A_i$ 的开矩体覆盖 $\{I_{ik}\}_{k\ge 1}$ 使得 
\[
\sum_{k\ge 1} \lvert I_{ik} \rvert < m^*(A_i) + \frac{\varepsilon}{2^i}
\] 

注意到 $\{I_{ik}\}_{i\ge 1}^{k\ge 1}$ 是 $\bigcup\limits_{i\ge1}A_i$ 的开覆盖，可得
\[
m^*\left(\bigcup_{i\ge1} A_i\right) \le \sum_{i\ge1}\sum_{k\ge1} \lvert I_k \rvert
\]
整合以上两式
\[
m^*\left(\bigcup_{i\ge1} A_i\right) < \sum_{i\ge1}m^*(A_i) + \varepsilon\sum_{i\ge1}\frac{1}{2^i}=\sum_{i\ge1}m^*(A_i) + \varepsilon
\]
对 $\varepsilon$ 求极限使其趋近于 $0$
\[
m^*\left(\bigcup_{i\ge1} A_i\right) \le \sum_{i\ge1}m^*(A_i)
\]
\ITEME
\item 平移不变形
\ITEME

\]

可以证明：不存在一个测度，既能测量 $\R^n$ 中的任意子集的大小，又兼具可数可加性。而可数可加性是我们想要的核心性质，有鉴于此，我们自然会想到：不妨把我们的研究对象限定为 $\R^n$ 中满足可数可加性的所有子集构成的集合。

\subsection{可测集}

不严格地讲，我们把在勒贝格外测度 $m^*(\cdot)$ 下，满足可数可加性的 $\R^n$ 中 的所有子集都称为可测集，否则为不可测集。注意到这一粗略定义在用来判断 $\R^n$ 中的任意子集是否为可测集时，根本不具备可操作性。自然地，我们想到通过最简单的情况：开方体上的有限可加性来定义可测集，希望以此来诱导出可数可加性。

\smallskip

\no {\bf 定义1：}

$E \subseteq \R^n$ 称为可测集，如果对任意开矩体 $I$ 都有
\[
m^*(I) = m^*(I\cap E) + m^*(I\cap E^c)
\]

为了使可测集的定义用起来更方便，我们希望把用来测试可测集的任意开矩体 $I$ 替换成任意子集 $T$，于是我们来到\\
\no {\bf 定义2：}

$E \subseteq \R^n$ 称为可测集，如果对 $\forall T \subseteq \R^n$ 都有
\[
m^*(T) = m^*(T\cap E) + m^*(T\cap E^c)
\]
其中的 $T$ 称为试验集。

下面我们证明这两种定义的等价性。\\
\pro 
定义2 $\Rightarrow$ 定义1: 显然。\\
定义1 $\Rightarrow$ 定义2：

由次可加性
\[
m^*(T) \le m^*(T\cap E) + m^*(T\cap E^c)
\]
对 $\forall \varepsilon > 0$ 存在 $T$ 的 $L$-覆盖使得
\[
\sum \lvert I_k \rvert < m^*(T) +\varepsilon
\]
再由单调性易知
\[
m^*(T\cap E) + m^*(T\cap E^c)\le m^*\left(\left(\bigcup\limits_k I_k\right)\cap E\right) + m^*\left(\left(\bigcup\limits_k I_k\right)\cap E^c\right)=m^*\left(\bigcup\limits_k \left(I_k\cap E\right) \right)+ m^*\left(\bigcup\limits_k \left(I_k\cap E^c\right)\right)
\]
由次可加性并经整理可得
\[
\mbox{上式右边} \le \sum_{k\ge1}\left[m^*\left( I_k\cap E\right) + m^*\left( I_k\cap E^c\right)\right] = \sum_{k\ge 1}\lvert I_k \rvert < m^*(T) +\varepsilon 
\]
于是
\[
m^*(T\cap E) + m^*(T\cap E^c)\le m^*(T) +\varepsilon 
\]
对上式取极限，让 $\varepsilon$ 趋近于 $0$，即得
\[
m^*(T) \ge m^*(T\cap E) + m^*(T\cap E^c)
\]
{\bf 注} 定义 2 中用于判断 $E$ 可测的条件 $m^*(T) = m^*(T\cap E) + m^*(T\cap E^c)$ 称为 Caratheodory 条件。由次可加性易知，$m^*(T) \le m^*(T\cap E) + m^*(T\cap E^c)$ 永远成立。故欲证该条件成立，只需证明 $m^*(T) \ge m^*(T\cap E) + m^*(T\cap E^c)$ 即可。

\subsubsection{零测集是可测集}

若 $m^*(E) = 0$，则 $E$ 称为零测集。\\
注意到，对任一零测集 $E$，由单调性易知 $ m^*(T\cap E) + m^*(T\cap E^c)\le m^*(E) +m^*(T)=m^*(T) $，从而 $E$ 是可测集。由此知 有理数集 $\mathbb{Q}$ 和康托三分集 $\mathcal{C}$ 均为可测集。

\bigskip

\EXP 设 $E_1, E_2$ 是 $\R^n$ 的子集。若 $E_1, E_2$ 能被 $\R^n$ 中一可测集 $E$ 分隔开 （即 $E_1, E_2$ 中一个含于 $E$  中，另一个含于 $E^c$ 中），则 $$m^*(E_1\cup E_2)=m^*(E_1)+m^*(E_2)$$ 特别地，若$E_1, E_2$ 是互不相交的可测集，则 $$m^*(E_1\cup E_2)=m^*(E_1)+m^*(E_2)$$

\pro 不妨设 $E_1\subseteq E, E_2\subseteq E^c$。由 Caratheodory 条件，我们有
$$m^*(E_1\cup E_2)=m^*((E_1\cup E_2)\cap E) + m^*((E_1\cup E_2)\cap E^c)= m^*(E_1)+m^*(E_2)$$
